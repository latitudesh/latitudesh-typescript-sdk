  bearer: process.env["LATITUDESH_BEARER"] ?? "",
});

  const result = await latitudesh.userProfile.update({
    id: "user_8PMvy2GL72uKKyXyANEnsjMQYMP",
    requestBody: {
      data: {
        id: "user_8PMvy2GL72uKKyXyANEnsjMQYMP",
        type: "users",
        attributes: {
          role: "collaborator",
        },
      },
    },
  });
